//Exemplo de Captura de Valor de Caixa de Texto // Captura o elemento de entrada de texto e o botão const minhaCaixaDeTexto = document.getElementById("minhaCaixaDeTexto"); const meuBotao = document.getElementById("meuBotao"); // Adiciona um ouvinte de evento ao botão para capturar o valor da caixa de texto e exibi-lo no console meuBotao.addEventListener("click", function() { const valorDaCaixaDeTexto = minhaCaixaDeTexto.value; console.log("O valor da caixa de texto é: " + valorDaCaixaDeTexto); });
